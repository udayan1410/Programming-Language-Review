1] stdio.h - Standard input/output functions

  • printf()
    •  - It writes formatted data to screen.
    Syntax:
    printf("Text & format specifier",list of arguments); //arguments are optional

    Example:
    
i]  printf("Hello World !!!"); 
		Output: Hello World !!!
		
ii] int a=5;
         printf("A=%d",a);
         Output: A=5


  • scanf()
    • - It reads formatted data from keyboard.
    Syntax:
    scanf("format specifier",&variable_name); //"&" is not given for %s(string) format specifier

    Example:
    
float m;
scanf("%f",m);
    IMP Note - All syntax in C language including printf() and scanf() functions are CASE-SENSITIVE.
2] string.h - String handling functions

  • strlen()
    • It calculates length of string.
    Syntax:
    strlen(string1);
    Example :
    			
int k;
k=strlen("PLR");
printf("%d",k);
Output: 3
  • strcpy()
    • It copies the string. Also copies the null character.
    Syntax:
    strcpy(string2,string1);
    Example :
    			
char s1[];
strcpy(s1,"This is SPARTAAA");
printf("%d",s1);
Output: This is SPARTAAA
  • strcat()
    • It concatenates i.e merges two strings.
    Syntax:
    strcat(string2,string1);
    Example :
    			
char S1[]="HELLO";
char S2[]="WORLD";
strcat(S1,S2);
printf(String after concatenation is:%c",S1);
Output:
String after concatenation is: HELLOWORLD
  • strupr()
    • It converts the string to Uppercase letters.
    Syntax:
    strupr(string1);
    Example :
    			
char s1[]="Doggy";
printf("%d",strupr(s1));
Output: DOGGY
  • strlwr()
    • It converts the string to lowercase letters.
    Syntax:
    strlwr(string1);
    Example :
    			
char s1[]="DOGGY";
printf("%d",strlwr(s1));
Output: doggy
3] math.h - Mathematical functions

  • ceil()
    • - It returns a value greater than or equal to x.
    Syntax:
    ceil(double x);

    Example :
    				
void main()
{
    float val1,val2;
    val1=1.2;
    val2=-2.8;
    printf("value1 = %.1lf\n",ceil(val1));
    printf("value2 = %.1lf\n",ceil(val2));
}
   Output: value1 = 2.0 
	value2 = -2.0


  • floor()
    • - It returns a value less than or equal to x.
    Syntax:
    floor(double x);

    Example :
    void main()
{
  float val1,val2;
  val1 = 1.2;
  val2 = -2.8;
  printf("value1 = %.1lf\n",floor(val1));
  printf("value2 = %.1lf\n",floor(val2));
}
  Output: value1 = 1.0 
   value2 = -3.0


  • log()
    • - It returns the natural logarithm (base-e logarithm) of x.
    Syntax:
    log(double x);

    Example :
    void main()
{
   double x, res;
   x=2.7;
   res=log(x);
   printf("log(%.4lf) = %.4lf",x,res);
}
Output: log(2.7000) = 0.9932


  • log10()
    • - It returns the common logarithm (base-10 logarithm) of x.
    Syntax:
    log10(double x);

    Example :
    void main()
{
   double x, res;
   x=10000;
   res=log10(x);
   printf("log(%.1lf) = %.1lf",x,res);
}
Output: log(10000.0) = 4.0


  • fmod()
    • - returns remainder when "x" is divided by "y"
    Syntax:
    fmod(double x, double y)

    Example :
    int main () 
{ 
   float a, b; 
   int c; 
   a = 8.2; 
   b = 5.7; 
   c = 3; 
   printf("Remainder of %f / %d is %lf\n", a, c, fmod(a, c)); 
   printf("Remainder of %f / %f is %lf\n", a, b, fmod(a, b)); 
     
   return(0); 
} 

Output: Remainder of 8.200000 / 3 is 2.200000
Remainder of 8.200000 / 5.700000 is 2.500000


  • sqrt()
    • - returns square root of a number
    Syntax:
    double sqrt(double x)

    Example :
    int main () 
{ 
  
   printf("Square root of %lf is %lf\n", 225.0, sqrt(225.0) ); 
   printf("Square root of %lf is %lf\n", 300.0, sqrt(300.0) ); 
     
   return(0); 
} 

Output: Square root of 225.000000 is 15.000000
Square root of 300.000000 is 17.320508


  • pow()
    • - returns x raised to power of y
    Syntax:
    double pow(double x, double y)

    Example :
    int main () 
{ 
   printf("Value 8.0 ^ 3 = %lf\n", pow(8.0, 3)); 
  
   printf("Value 3.05 ^ 1.98 = %lf", pow(3.05, 1.98)); 
     
   return(0); 
} 

Output:Value 8.0 ^ 3 = 512.000000
Value 3.05 ^ 1.98 = 9.097324
Importance of main() function -

Every C program has a primary (main) function that must be named main. It is important that every c program has exactly one main() function or else the program will show an error. The main function serves as the starting point and the end point for program execution. It usually controls program execution by directing the calls to other functions in the program.
Syntax:
datatype main() { .... }
OR
datatype main(int argc, char *argv[], char *envp[]) //For command-line arguments { .... }
